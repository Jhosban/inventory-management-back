cURL para crear loan
curl --location 'http://localhost:8080/api/loans' \
--data '{
    "itemId": 1,
    "quantity": 1,
    "startDate": 1749584070,
    "endDate": 1750001000000,
    "requestedBy": "John"
}'

cURL PARA CANCELAR UN LOAN
curl --location --request PUT 'http://localhost:8080/api/loans/1/cancel' \
--header 'Content-Type: application/json' \
--data '1'

cURL para obtener loans
curl --location 'http://localhost:8080/api/loans'
